Understanding JS Syntax

Comments:

// This is a single-line comment

/*
   This is a
   multi-line comment
*/

Variables:

var x = 5; // Declaring a variable using var
let y = 10; // Declaring a variable using let (block-scoped)
const PI = 3.14; // Declaring a constant

Data Types:

var num = 5; // Number
var str = "Hello"; // String
var isTrue = true; // Boolean
var arr = [1, 2, 3]; // Array
var obj = { name: "John", age: 30 }; // Object

Operators:

var sum = 5 + 3; // Addition
var difference = 10 - 4; // Subtraction
var product = 6 * 2; // Multiplication
var quotient = 8 / 2; // Division
var remainder = 10 % 3; // Modulus

var x = 5;
x += 3; // Same as x = x + 3

var isEqual = (5 === "5"); // Comparison (strict equality)

var result = (5 > 3) && (10 < 20); // Logical AND

Functions:

function greet(name) {
    console.log("Hello, " + name + "!");
}

greet("John"); // Output: Hello, John!
